'''--- src/lib.rs ---
use near_sdk::borsh::{self, BorshDeserialize, BorshSerialize};
use near_sdk::{env, near_bindgen};

near_sdk::setup_alloc!();

#[derive(Default, BorshDeserialize, BorshSerialize)]
pub struct Owner {
    name: String,
    balance: u32,
}

#[derive(Default, BorshDeserialize, BorshSerialize)]
pub struct Employer {
    name: String,
    balance: u32,
}

#[derive(Default, BorshDeserialize, BorshSerialize)]
pub struct Gov {
    name: String,
    balance: u32,
}

#[near_bindgen]
#[derive(Default, BorshDeserialize, BorshSerialize)]
pub struct Contract {
    owner: Owner,
    employer: Employer,
    gov: Gov,
}

impl Contract {
    pub fn get_owner_balance(&self) -> u32 {
        self.owner.balance
    }

    pub fn charge(&mut self, amount: u32) {
        self.employer.balance -= amount;
        self.owner.balance += amount;
        let log_message = format!("Charging {} from Employer", amount);
        env::log(log_message.as_bytes());
    }
}

#[cfg(test)]
mod test {
    use super::*;
    use near_sdk::MockedBlockchain;
    use near_sdk::{testing_env, VMContext};

    fn get_context(input: Vec<u8>, is_view: bool) -> VMContext {
        VMContext {
            current_account_id: "alice.testnet".to_string(),
            signer_account_id: "jomsox.testnet".to_string(),
            signer_account_pk: vec![0, 1, 2],
            predecessor_account_id: "jane.testnet".to_string(),
            input,
            block_index: 0,
            block_timestamp: 0,
            account_balance: 0,
            account_locked_balance: 0,
            storage_usage: 0,
            attached_deposit: 0,
            prepaid_gas: 10u64.pow(18),
            random_seed: vec![0, 1, 2],
            is_view,
            output_data_receivers: vec![],
            epoch_height: 19,
        }
    }

    #[test]
    fn charge() {
        let context = get_context(vec![], false);
        testing_env!(context);
        let mut contract = Contract {
            owner: Owner { name: String::from("Jose Sosa"), balance: 10 },
            employer: Employer { name: String::from("Near Co"), balance: 10000 },
            gov: Gov { name: String::from("SAT"), balance: 0 },
        };
        contract.charge(55);
        println!("Owner new balance: {}", contract.get_owner_balance());
        assert_eq!(65, contract.get_owner_balance());
    }
}
