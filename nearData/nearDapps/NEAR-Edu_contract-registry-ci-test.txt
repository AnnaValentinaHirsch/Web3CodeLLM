          git checkout "$(<../checkout.txt)"

      - store_artifacts:
          path: /tmp/artifacts

          rm -rf /tmp/artifacts
          mkdir -p /tmp/artifacts

          no_output_timeout: 5m
      - store_artifacts:
          path: /tmp/artifacts
          destination: out
